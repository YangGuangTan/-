   454→function useAnimatedNumber(target: number, duration = 1500, decimals = 0) {
   455→  const [current, setCurrent] = useState(0)
   456→  const startTime = useRef<number | null>(null)
   457→  const rafId = useRef<number | null>(null)
   458→  const prevTarget = useRef(target)
   459→
   460→  useEffect(() => {
   461→    startTime.current = null
   462→    const startValue = prevTarget.current === target ? current : prevTarget.current
   463→    prevTarget.current = target
   464→    const animate = (ts: number) => {
   465→      if (!startTime.current) startTime.current = ts
   466→      const p = Math.min((ts - startTime.current) / duration, 1)
   467→      const eased = 1 - Math.pow(1 - p, 3)
   468→      setCurrent(Number((startValue + (target - startValue) * eased).toFixed(decimals)))
   469→      if (p < 1) rafId.current = requestAnimationFrame(animate)
   470→    }
   471→    rafId.current = requestAnimationFrame(animate)
   472→    return () => { if (rafId.current) cancelAnimationFrame(rafId.current) }
   473→  }, [target, duration, decimals])
   474→  return current
   475→}

   477→function CustomTooltip({ active, payload, label }: TooltipProps<number, string>) {
   478→  if (!active || !payload?.length) return null
   479→  return (
   480→    <div className="custom-tooltip" style={{ padding: '10px 14px', borderRadius: 10 }}>
   481→      {label && <div style={{ color: '#475569', marginBottom: 6, fontSize: 12, fontWeight: 600 }}>{label}</div>}
   482→      {payload.map((e, i) => (
   483→        <div key={i} style={{ display: 'flex', alignItems: 'center', gap: 6, marginBottom: 3 }}>
   484→          <div style={{ width: 8, height: 8, borderRadius: 3, backgroundColor: e.color || '#0891b2' }} />
   485→          <span style={{ color: '#64748b', fontSize: 12 }}>{e.name}:</span>

   487→        </div>
   488→      ))}
   489→    </div>
   490→  )
